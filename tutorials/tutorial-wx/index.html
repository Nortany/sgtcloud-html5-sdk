sdk微信使用教程

		var _appId = 'your_appId';	//用户标识，用于初始化

		//第一步，初始化sgt
		sgt.init({
			appId: _appId;
		});

		//第二步，调用微信中控服务
		sgt.WxCentralService.getSignature(function(result, data) {
			if (result) {
				//成功返回签名信息，执行微信服务
				doWxApi(data);
			}
		});

		//第三步，执行微信api服务
		function doWxApi(data){
			//微信配置
			wx.config({
				appId: "微信公众号appId"	//需要在后台进行绑定
				timestamp: data.timestamp,
				nonceStr: data.noncestr,
				signature: data.signature
			})

			//微信其他api
			wx.ready();
		}
