Course 3分钟搭建一个排行榜

Step 1:

让我们先来把SgtApi初始化一下 
        SgtApi.init({
            "AppId": 'html5_demo2015',//设置应用标识
            "ChannelId": ''//设置渠道标识
        });

Step 2:

接着我们来登录用户 
        SgtApi.AccountService.quickLogin(function (ret, data) {
            if (ret) {
                /*
                 success do something ...
                 */
            } else {
                throw new Error('快速登录失败');
            }
        });

Step 3:

现在我们就可以来创建一个角色了 
        var playerData;
        if (SgtApi.PlayerService.init()) {
            //创建角色
            var players = new SgtApi.entity.Player();
            players.name = 'woooooooo';
            players.customId = "player0";
            players.equip = "";
            players.level = '5';
            players.gender = '1';
            players.money = '998';

            SgtApi.PlayerService.create(players, function (ret, data) {
                if (ret) {
                    //成功
                    playerData = data;
                    throw new Error('create player success');
                } else {
                    //失败
                    throw new Error('create player fail');
                }
            })
        } else {
            throw new Error("初始化角色接口失败");
        }
